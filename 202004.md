    function merge(arr1,arr2){
        let arrAll=[];
        var n=arr1&&arr1.length;
        var m=arr2&&arr2.length;
        let i=0;let j=0;
        while(i<n&&j<m){
            if(arr1[i]<arr2[j]){
                arrAll.push(arr1[i]);
                i++;
            }else{
                arrAll.push(arr2[j++]);
            }
        }
        while (i<n){
            arrAll.push(arr1[i++]);
        }

            arrAll.push(arr2[j++])

    function getMid(num){
        let top=num;
        let bottom=1;
        while(Math.abs((top+bottom)/2*(top+bottom)/2-num)>Math.abs(-1e-6)){
            let mid=(top+bottom)/2;
            if(mid*mid>num){
                top=mid
            }else{
                bottom=mid
            }
        }
        return (top+bottom)/2
    }
